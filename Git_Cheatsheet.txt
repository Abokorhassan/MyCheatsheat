git stash -u ---> for new files and all

git stash applly 0 ---> bring back the last stash 
git stash show 0 ---> show changes the stashed files.
git stash show -p 0 --> detailed show.
git stash push -m "update" --> provide name or desc
git stash branch new-branch 0 --> make branch from prev stash.
git stash pop ---> get the last stash.
git stash drop 0 ---> remove last stash.
git stash clear --> remove all stash.
